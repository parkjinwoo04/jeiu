import asyncio import re from playwright.async_api import async_playwright from typing import List, Dict # 타임아웃 설정 (10초) TIMEOUT = 10000 class NaverMapRestaurantCrawler: def __init__(self, headless: bool = False): self.headless = headless self.launch_options = { "headless": self.headless, "args": ["--no-sandbox", "--disable-dev-shm-usage", "--disable-blink-features=AutomationControlled"], } self.context_options = { "viewport": {"width": 1920, "height": 1080}, "user_agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/121.0.0.0 Safari/537.36", } async def _perform_search(self, page, search_query: str): """네이버 지도 접속 및 검색어 입력""" print(f"1. 네이버 지도 접속 중... (검색어: {search_query})") await page.goto("https://map.naver.com/", wait_until="domcontentloaded") search_input = await page.wait_for_selector("input.input_search", state="visible", timeout=TIMEOUT) await search_input.click() await search_input.fill(search_query) await search_input.press("Enter") # 검색 결과 iframe이 뜰 때까지 대기 await page.wait_for_selector("iframe#searchIframe", state="visible", timeout=TIMEOUT) async def _get_search_frame(self, page): """검색 결과 iframe 가져오기""" iframe_element = await page.query_selector("iframe#searchIframe") return await iframe_element.content_frame() async def _scroll_to_load_all(self, frame): """모든 식당이 로드될 때까지 스크롤 내리기""" print("2. 검색 결과 리스트 스크롤 내리는 중...") previous_count = 0 no_change_count = 0 max_no_change = 3 while True: current_restaurants = await frame.query_selector_all("li.UEzoS") current_count = len(current_restaurants) if current_count == previous_count: no_change_count += 1 if no_change_count >= max_no_change: print(" => 모든 식당 로드 완료!") break else: no_change_count = 0 previous_count = current_count # 스크롤 엘리먼트를 찾아 맨 아래로 내림 await frame.evaluate(""" () => { const scrollContainer = document.querySelector('.Ryr1F') || document.body; if (scrollContainer) { scrollContainer.scrollTop = scrollContainer.scrollHeight; } } """) await asyncio.sleep(1.5) async def _extract_basic_info(self, restaurant): """식당 이름 및 카테고리 추출""" name_elem = await restaurant.query_selector("span.TYaxT") name = await name_elem.inner_text() if name_elem else "이름 없음" category_elem = await restaurant.query_selector("span.KCMnt") category = await category_elem.inner_text() if category_elem else "분류 없음" return name, category async def _extract_place_id(self, restaurant, page): """식당을 클릭하여 URL에서 place_id 추출""" link_elem = await restaurant.query_selector("a.place_bluelink") if not link_elem: return None await link_elem.click() try: # URL에 place가 포함될 때까지 대기 await page.wait_for_url(lambda url: "/place/" in url, timeout=5000) match = re.search(r"/place/(\d+)", page.url) return match.group(1) if match else None except: return None async def _extract_address_info(self, place_id: str, context): """독립된 새 탭을 열어 주소 정보 추출""" place_detail_url = f"https://pcmap.place.naver.com/place/{place_id}" detail_page = await context.new_page() try: await detail_page.goto(place_detail_url, wait_until="domcontentloaded") await detail_page.wait_for_selector("span.LDgIH", timeout=5000) address_elem = await detail_page.query_selector("span.LDgIH") address = await address_elem.inner_text() if address_elem else "주소 없음" return address except: return "주소 로딩 실패" finally: await detail_page.close() async def crawl_single_page(self, search_query: str) -> List[Dict]: """전체 크롤링 프로세스 실행""" async with async_playwright() as p: browser = await p.chromium.launch(**self.launch_options) context = await browser.new_context(**self.context_options) page = await context.new_page() # 이미지 로딩을 차단하여 속도 향상 await page.route("**/*.{png,jpg,jpeg,gif,svg,webp}", lambda route: route.abort()) results = [] try: await self._perform_search(page, search_query) frame = await self._get_search_frame(page) if not frame: return results await self._scroll_to_load_all(frame) restaurants = await frame.query_selector_all("li.UEzoS") print(f"3. 총 {len(restaurants)}개의 식당 발견. 데이터 추출 시작...") for idx, restaurant in enumerate(restaurants, 1): name, category = await self._extract_basic_info(restaurant) place_id = await self._extract_place_id(restaurant, page) address = "N/A" if place_id: address = await self._extract_address_info(place_id, context) print(f" [{idx}/{len(restaurants)}] {name} ({category}) - {address}") results.append({ "place_id": place_id, "name": name, "category": category, "address": address }) # 다시 리스트를 보기 위해 뒤로 가기 await page.go_back() await asyncio.sleep(0.5) except Exception as e: print(f"크롤링 중 오류 발생: {e}") finally: await browser.close() return results async def main(): search_query = input("식당 크롤링 할 위치를 입력하세요 (예: 공덕역 식당): ") if not search_query: search_query = "공덕역 식당" crawler = NaverMapRestaurantCrawler(headless=False) results = await crawler.crawl_single_page(search_query) print("\n========= 수집 완료 결과 =========") for res in results: print(f"ID: {res['place_id']} | 이름: {res['name']} | 분류: {res['category']} | 주소: {res['address']}") if __name__ == "__main__": asyncio.run(main())
